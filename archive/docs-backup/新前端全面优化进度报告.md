### 1. 使用响应式Hooks

```vue
<script setup lang="ts">
import { useDeviceType, usePageLoading } from '@/hooks/web'
import { formatDateTime } from '@/utils/common'

const { isMobile, breakpoint } = useDeviceType()

// 根据设备类型显示不同内容

  isMobile.value ? 'card' : 'table'
)
</script>
```

### 2. 使用请求重试和缓存

```typescript
import { executeWithRetry, executeWithCache } from '@/utils/http/alova/enhancers'
import { getCompetitionList } from '@/api/competition/info'

// 带重试的请求
const fetchData = async () => {
  const method = getCompetitionList({ page: 1, limit: 20 })
  const res = await executeWithRetry(method, {
    maxRetries: 3,
    retryDelay: 1000,
  })
  tableData.value = res.page.list
}

// 带缓存的请求
const fetchCachedData = async () => {
  const method = getCompetitionList({ page: 1, limit: 20 })
  const res = await executeWithCache(method, {
    cacheKey: 'competition_list_page_1',
    ttl: 5 * 60 * 1000, // 5分钟
  })
  tableData.value = res.page.list
}
```

### 3. 使用表单验证

```vue
<script setup lang="ts">
import { requiredEmail, requiredPhone, passwordStrength } from '@/utils/formValidator'

const rules = {
  email: requiredEmail(),

  password: [required(), passwordStrength()],
}
</script>

<template>

    <NFormItem label="邮箱" path="email">
      <NInput v-model:value="form.email" />
    </NFormItem>
  </NForm>
</template>
```
