public boolean validateFileContent(byte[] fileBytes) {
    if (fileBytes.length < 4) return false;
    
    // PDF文件魔数：%PDF
    if (fileBytes[0] == '%' && fileBytes[1] == 'P' && 
        fileBytes[2] == 'D' && fileBytes[3] == 'F') {
        return true;
    }
    
    // ZIP文件魔数：PK

}
